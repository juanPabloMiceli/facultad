 68         int mayor = max(i, j);

198     Lock enqueueLock;

201     int contadorDequeue = 0;

204         if(getSize() >= capacity){ // Muito lento
205             return;

208         //
209         contadorEnqueue++;
210         //
211         enqueueLock.release();
212     }

215         if(getSize() == 0){ // Muito lento
216             return;
217         }
218         dequeueLock.lock();
219         //
220         contadorDequeue++;
221         //
222         dequeueLock.release();
223

226     int getSize(){
227         enqueueLock.lock();
228         dequeueLock.lock();
229         int delta = contadorEnqueue - contadorDequeue;
230         size += delta

232         contadorDequeue = 0;
233         enqueueLock.release();
234         dequeueLock.release();
235         return size;

240 Class BoundedStack {
241 
242     final int capacity;
243     int size = 0;
244     Node head = Node(INT_MIN);
245 
246     void push(Object o){
247         head.lock();
248         try{
249             if(capacity >= size) {
250                 throw new FullException();
251             }
252             Node next = head.next();
253             Node newNode = new Node(o);
254             head.next() = newNode;
255             newNode.next() = next;
256             size++;
257         } finally {
258             head.release();
259         }
260     }
261 
262     // Alcanza con solo lockear head porque es exactamente lo mismo que un monitor (para arrancar si o si tenes que
263     // tener acceso a head)
264     Object pop(){
265         head.lock();
266         try {
267             if(head.next() == null){
268                 throw new EmptyException();
269             }
270             Node current = head.next();
271             Object res = current.item;
272             head.next() = current.next();
273             size--;
274         } finally {
275             head.release();
276         }
277         return res;
278     }
279 }

323 Class ABAStack {
324 
325     private AtomicReference top = new AtomicReference(null);
326 
327     boolean tryPush(Node node) {
328         Node oldTop = top.get();
329         node.next = oldTop;
330         return top.compareAndSet(oldTop, node);
331     }
332 
333     void push(Object o){
334         Node node = Memory.new(o);
335         while(true) {
336             if(tryPush(node)) {
337                 return;
338             } else {
339                 continue;
340             }
341         }
342     }
343 
344     Node tryPop() {
345         Node oldTop = top.get();
346         Node newTop = oldTop.next;
347         if(top.compareAndSet(oldTop, newTop)){
348             return oldTop;
349         }
350         return null;
351     }
352 
353     Object pop() {
354         while(true){
355             Node result = tryPop();
356             if(result != null){
357                 return result.item;
358             }
359         }
360     }
361

364 Class ABAFreeStack {
365 
366     private AtomicStampedReference head = new AtomicStampedReference(null, 0);
367     int[] stamp = new int[0];
368 
369     boolean tryPush(Node node) {
370         Node oldHead = head.get(stamp);
371         node.next = oldHead;
372         return head.compareAndSet(oldHead, node, stamp[0], stamp[0] + 1);
373     }
374 
375     void push(Object o){
376         Node node = Memory.new(o);
377         while(true) {
378             if(tryPush(node)) {
379                 return;
380             } else {
381                 continue;
382             }
383         }
384     }
385 
386     Node tryPop() {
387         Node oldHead = head.get(stamp);
388         Node newHead = oldHead.next;
389         if(head.compareAndSet(oldHead, newHead, stamp[0], stamp[0]+1)){
390             return oldHead;
391         }
392         return null;
393     }
394 
395     Object pop() {
396         while(true){
397             Node result = tryPop();
398             if(result != null){
399                 Object res = result.item;
400                 Memory.free(result);
401                 return res;
402             }
403         }
404     }
